Фрактал, один из двумерных аналогов множества Кантора. Полностью самоподобен и строиться методом хаоса. 
    let canvas2 = document.getElementById('canvas2')
    if (canvas2) {
      let ctx2 = canvas2.getContext("2d")

      let fx = 250
      let fy = 480
      let x, y

      for (let i = 0; i < 10000; i++) {
        switch (Math.floor(Math.random() * 3)) {
          case 0:
            x = (fx + 20) / 2
            y = (fy + 480) / 2
          break;
          case 1:
            x = (fx + 250) / 2
            y = (fy + 60) / 2
          break;
          case 2:
            x = (fx + 480) / 2
            y = (fy + 480) / 2
          break;
        }
        fx = x
        fy = y
        ctx2.fillRect(x, y, 1, 1)
      }
    }
